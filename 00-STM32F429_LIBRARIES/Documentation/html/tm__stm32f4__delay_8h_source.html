130 static __INLINE void Delay(uint32_t micros) {
131 #if defined(TM_DELAY_TIM)
132  volatile uint32_t timer = TM_DELAY_TIM->CNT;
133 
134  do {
135  /* Count timer ticks */
136  while ((TM_DELAY_TIM->CNT - timer) == 0);
137 
138  /* Increase timer */
139  timer = TM_DELAY_TIM->CNT;
140 
141  /* Decrease microseconds */
142  } while (--micros);
143 #else
144  uint32_t amicros;
145 
146  /* Multiply micro seconds */
147  amicros = (micros) * (mult);
148 
149  #ifdef __GNUC__
150  if (SystemCoreClock == 180000000 || SystemCoreClock == 100000000) {
151  amicros -= mult;
152  }
153  #endif
154 
155  /* If clock is 100MHz, then add additional multiplier */
156  /* 100/3 = 33.3 = 33 and delay wouldn't be so accurate */
157  #if defined(STM32F411xE)
158  amicros += mult;
159  #endif
160 
161  /* While loop */
162  while (amicros--);
163 #endif /* TM_DELAY_TIM */
164 }
